\section{Research Hypothesis}
%% ═══════════════════════════════════════════════════════════

\begin{frame}{Central Hypothesis}
\presenterphoto{sienna_ml}
  \vspace{-0.7em}
  \begin{block}{Research Question}
    Do \alert{individual football player} rolling statistics add predictive
    signals beyond team-level aggregates?
  \end{block}

  \vspace{0.3em}

  \textbf{Goal:} predict Ligue 1 match outcomes (home win, draw, away win)
  using team and player data. Train on 2018--2022; test on 2023/24.

  \vspace{0.5em}

  \textbf{Three feature regimes evaluated:}

  \vspace{-0.6em}
  \begin{columns}[T]
    \column{0.32\textwidth}
    \begin{block}{Regime A}
      \small Team-level aggregates\\[0.2em]
      29 features
    \end{block}

    \column{0.32\textwidth}
    \begin{block}{Regime B}
      \small Team + FBref match statistics\\[0.2em]
      50 features
    \end{block}

    \column{0.32\textwidth}
    \begin{alertblock}{Regime C}
      \small Full set incl.\ \alert{player-rolling} features\\[0.2em]
      76 features
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}[shrink=5]{Feature Engineering}
\presenterphoto{sienna_ml}
  \vspace{0.3em}
  \begin{center}
  \small
  {\setlength{\tabcolsep}{6pt}\renewcommand{\arraystretch}{1.15}
  \begin{tabularx}{0.95\linewidth}{lXl}
    \toprule
    Source & Features engineered & Regimes \\
    \midrule
    Understat + Transfermarkt
      & Rolling xG, goals, wins, points; squad value, age
      & A, B, C \\
    FBref team-level
      & Rolling possession, formation, goals $+$ assists per 90 (team)
      & B, C \\
    \alert{Player-level}
      & \alert{Rolling shots, shots on target, defensive actions, goals, assists per 90}
      & \alert{C only} \\
    \bottomrule
  \end{tabularx}}
  \end{center}

  \vspace{0.6em}
  \footnotesize
  \begin{itemize}\setlength\itemsep{0.2em}
    \item \textbf{Anti-leakage:} 5-match rolling window with \texttt{shift(1)} ---
          current match never in its own features
    \item \textbf{Bookmaker probabilities excluded} from model inputs ---
          kept as external baseline only (including them would be circular)
    \item \textbf{Player rolling stats are personal, not club-based} ---
          if a player transfers, their form history carries over
  \end{itemize}
\end{frame}

\begin{frame}{Player-Level Aggregation}
\presenterphoto{sienna_ml}
  \begin{columns}[T]
    \column{0.50\textwidth}
    Each player's 5-match rolling stat is aggregated across the starting XI,
    \textbf{weighted by minutes played}:
    \[
      \hat{s}_{\text{team}} = \frac{\sum_{p \in \text{XI}} \text{min}_p \cdot \text{roll}_{p}}{\sum_{p \in \text{XI}} \text{min}_p}
    \]
    \vspace{0.2em}
    Each stat then becomes three features:\\[0.3em]
    \texttt{home\_pl\_*} \quad \texttt{away\_pl\_*} \quad \texttt{net\_pl\_*}

    \column{0.46\textwidth}
    \begin{block}{Stats tracked per player}
      \small
      \begin{tabular}{ll}
        \toprule
        Feature & Meaning \\
        \midrule
        \texttt{sh\_p90}       & shots / 90 \\
        \texttt{sot\_p90}      & shots on target / 90 \\
        \texttt{def\_p90}      & tackles + int.\ / 90 \\
        \texttt{gls\_p90}      & goals / 90 \\
        \texttt{ast\_p90}      & assists / 90 \\
        \texttt{starter\_mins} & minutes regularity \\
        \texttt{pct\_fw}       & fraction forwards \\
        \texttt{depth\_sh}     & starter vs bench gap \\
        \bottomrule
      \end{tabular}
    \end{block}
  \end{columns}
\end{frame}

\section{Feature Selection}
%% ═══════════════════════════════════════════════════════════

\begin{frame}{Feature Selection — Four-Method Protocol}
\presenterphoto{ioana_ml}
  \footnotesize Each of the 73 features was scored by \textbf{four independent methods},
  percentile-normalised (0--1), and averaged into a \textbf{composite score}.

  \vspace{0.25em}

  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{1. Filter Method}
      \footnotesize Simple idea: if I know this feature, do I get a better clue
      about the match result? Good for catching patterns that are not just straight lines.
    \end{block}
    \vspace{0.15em}
    \begin{block}{2. Wrapper Method}
      \footnotesize Start with many features, remove the weakest ones step by step,
      and keep the set that gives the best validation score.
      Best subset: \textbf{26 features}.
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{{\scriptsize 3. Embedded Method — Lasso Regression}}
      \footnotesize A linear model with a penalty that pushes unhelpful features
      down to exactly zero. 17 features were removed; \textbf{0 player features} were.
    \end{block}
    \vspace{0.15em}
    \begin{block}{{\scriptsize 4. Embedded Method — Decision Tree}}
      \footnotesize Let a forest of trees rank which features it uses most to make
      good splits. This gives a second opinion that broadly matches RFECV.
    \end{block}
  \end{columns}

  \vspace{0.2em}
  \footnotesize\textit{Based on Lecture 9's taxonomy of feature selection methods: filter, wrapper, and embedded methods. Information gain is developed in Lecture 8; Lasso regularization is covered in Lecture 4; and decision-tree / random-forest feature importance is connected to Lecture 8.}\\[0.2em]
  \footnotesize\textit{Temporal CV (TimeSeriesSplit, 5 folds) throughout — data never shuffled across time.}
\end{frame}

  \end{alertblock}

  \begin{alertblock}{\scriptsize Log-loss \emph{increases} with more features --- the central diagnostic of the project.}
  \end{alertblock}
\end{frame}


%% ── Slide 3: Limits & Diagnostic ───────────────────────────────
\begin{frame}[t,shrink=8]{Linear Baseline — Limits \& Diagnostic}
\presenterphoto{ava_ml}
  \scriptsize \textbf{Why did adding features hurt?} Two compounding problems explain it.
  \vspace{0.3em}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{1. Multicollinearity}
      \scriptsize Team-level and player-level features measure overlapping things (a team's xG is essentially a function of its players' xG). Linear models cannot decide where to attribute the weight --- the optimisation has multiple equivalent solutions, leading to \alert{unstable coefficients}.
    \end{block}
    \column{0.48\textwidth}
    \begin{block}{2. Curse of dimensionality}
      \scriptsize Only \alert{417 training matches} against up to \alert{76 features} --- a samples-to-features ratio of $\sim$5.5. Rule of thumb: linear models need 10--20$\times$ more samples than features. Below that, they \textbf{overfit the noise} rather than learn the signal.
    \end{block}
  \end{columns}
  \vspace{0.3em}
  \begin{block}{This negative result is what motivated the tree models}
    \scriptsize Tree-based models handle correlated features natively --- each split uses only one feature at a time, so they're \emph{locally low-dimensional}. \textbf{Tree models flip the trend} (RF: 1.26 $\to$ 1.06 $\to$ 1.05). Same data, same protocol, opposite outcome --- which Samuel will detail next.
  \end{block}
  \vspace{0.2em}
  \scriptsize\textit{\textbf{Lesson:} model class matters more than feature count. The player-level signal exists --- but linear models cannot surface it.}
\end{frame}

%% ── KNN: how we chose k ────────────────────────────────────────
\begin{frame}{KNN — Choosing $k$}

  \footnotesize $k$ was not hand-picked. The full grid
  $k \in \{5, 10, 15, 20, 30, 50\} \times$ \{uniform, distance\}
  $\times$ \{Manhattan, Euclidean\} was searched under the same
  \texttt{TimeSeriesSplit(5)} protocol and the lowest-log-loss
  combination was kept.

  \vspace{0.4em}
  \begin{columns}[T]
    \column{0.5\textwidth}
    \begin{block}{Why those bounds?}
      \scriptsize
      \textbf{MIN $k=5$:} smaller $k$ gives degenerate probabilities
      (denominators of 2--3 only output $\{0, 0.33, 0.5, 0.66, 1\}$)
      — log-loss punishes that hard.\\[0.3em]
      \textbf{MAX $k=50$:} the smallest CV training fold is $\sim$75 rows;
      sklearn requires $k \le$ fold size. We initially tried $k=100$
      and got runtime warnings, so we capped at 50.
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{Winners (per regime)}
      \scriptsize
      \setlength{\tabcolsep}{4pt}
      \begin{tabular}{lcc}
        Regime & $k$ & metric \\
        \hline
        A: team   & 50 & Euclidean ($p{=}2$) \\
        B: +FBref & 50 & Manhattan ($p{=}1$) \\
        C: +player& 50 & Manhattan ($p{=}1$) \\
      \end{tabular}\\[0.4em]
      All three regimes \textbf{independently} picked $k=50$ with
      distance-weighting. The Minkowski exponent drifted from
      $p{=}2$ to $p{=}1$ as features grew — consistent with the
      textbook result that L1 is more robust in higher dimensions.
    \end{block}
  \end{columns}

  \vspace{0.4em}
  \footnotesize\textbf{Why $k=50$ makes domain sense:} distance-weighting
  means the closest 5--10 historical matchups still drive the
  prediction; the further 40 just provide a regularising prior.
  Football is noisy, so averaging $\sim$12\% of the historical sample
  smooths randomness without washing out the matchup-specific signal.
\end{frame}

\section{Key Findings}
%% ═══════════════════════════════════════════════════════════

\begin{frame}{Key Findings}
\presenterphoto{sam_ml}
  \small
  \begin{enumerate}
    \item Player-level rolling statistics add real predictive signal beyond team form and bookmaker odds.
    \item Feature selection backs that up: \texttt{net\_pl\_starter\_mins} and
    \texttt{net\_pl\_pct\_fw} rank highly across all four methods, and RFECV keeps
    10 player features in its final subset.
    \item Tree models give the clearest proof of the player signal: as we move from
    team-only features to the full player regime, every tree model improves, while
    logistic regression gets worse under the extra correlation and dimensionality.
    \item The best tree result is the RF + XGBoost ensemble on the full player regime
    at \textbf{1.04} log-loss and 50.6\% accuracy, so ensembling still extracts a small final gain.
    \item KNN is a useful surprise: with $k{=}50$ it ties the stronger models
    (\textbf{1.01} in regime B, 1.02 in regime C), suggesting that averaging similar
    past match states works well when the sample is modest and noisy.
    \item Six different algorithms finish within about 0.03 log-loss of one another,
    which points to an information ceiling in historical aggregate stats; even so,
    the full pipeline is fully reproducible with only public data and closes about
    one-third of the gap from uniform guessing to Understat's 0.91 benchmark.
  \end{enumerate}
\end{frame}
