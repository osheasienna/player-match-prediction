\section{Research Hypothesis}
%% ═══════════════════════════════════════════════════════════

\begin{frame}{Central Hypothesis}
\presenterphoto{sienna_ml}
  \vspace{-0.7em}
  \begin{block}{Research Question}
    Do \alert{individual football player} rolling statistics add predictive
    signals beyond team-level aggregates?
  \end{block}

  \vspace{0.3em}

  \textbf{Goal:} predict Ligue 1 match outcomes (home win, draw, away win)
  using team and player data. Train on 2018--2022; test on 2023/24.

  \vspace{0.5em}

  \textbf{Three feature regimes evaluated:}

  \vspace{-0.6em}
  \begin{columns}[T]
    \column{0.32\textwidth}
    \begin{block}{Regime A}
      \small Team-level aggregates\\[0.2em]
      29 features
    \end{block}

    \column{0.32\textwidth}
    \begin{block}{Regime B}
      \small Team + FBref match statistics\\[0.2em]
      50 features
    \end{block}

    \column{0.32\textwidth}
    \begin{alertblock}{Regime C}
      \small Full set incl.\ \alert{player-rolling} features\\[0.2em]
      76 features
    \end{alertblock}
  \end{columns}
\end{frame}

\begin{frame}[t]{Pipeline \& Dataset}
\presenterphoto{sienna_ml}
\vspace{-0.4em}
  \begin{columns}[T]

    \column{0.54\textwidth}
    \textbf{How we built it:}\\[0.3em]
    \footnotesize
    \begin{enumerate}\setlength\itemsep{0.3em}
      \item Match team names across data sources
      \item Reshape each match into one row per team, ensuring home and away sides are handled symmetrically
      \item Construct a 5-match rolling window using only prior games, so the current match never contributes to its own features
      \item For Regime C, compute per-90 player statistics and average them across the starting eleven, weighted by minutes played
      \item Remove the first 439 matches because they do not yet have enough prior history for rolling features
    \end{enumerate}

      \item \alert{Understat / Transfermarkt: zero missingness}
    \end{itemize}

\begin{frame}[shrink=15]{Feature Engineering}
\presenterphoto{sienna_ml}
  \begin{columns}[T]
    \column{0.52\textwidth}
    \textbf{Three feature layers:}\\[0.2em]
    \footnotesize
    {\setlength{\tabcolsep}{4pt}\renewcommand{\arraystretch}{1.05}
    \begin{tabularx}{\linewidth}{lXl}
      \toprule
      Source & Features engineered & Regimes \\
      \midrule
      Understat + TM
        & xG, goals, wins, points; value, age
        & A, B, C \\
      FBref team
        & Possession, goals $+$ assists per 90
        & B, C \\
      \alert{Player}
        & \alert{Shots, shots on target, tackles + interceptions, goals, assists per 90}
        & \alert{C} \\
      \bottomrule
    \end{tabularx}}

    \vspace{0.4em}
    \footnotesize\textbf{Player aggregation (Regime C):}
    \[
      \hat{s}_{\text{team}} = \frac{\sum_{p \in \text{XI}} \text{min}_p \cdot \text{roll}_{p}}{\sum_{p \in \text{XI}} \text{min}_p}
    \]
    \column{0.44\textwidth}
    \begin{block}{Stats tracked per player}
      \scriptsize
      \begin{tabularx}{\linewidth}{lX}
        \toprule
        Feature & Meaning \\
        \midrule
        \texttt{sh\_p90}       & shots per 90 min \\
        \texttt{sot\_p90}      & shots on target per 90 \\
        \texttt{def\_p90}      & tackles + interceptions per 90 \\
        \texttt{gls\_p90}      & goals per 90 min \\
        \texttt{ast\_p90}      & assists per 90 min \\
        \texttt{starter\_mins} & average minutes per game \\
        \texttt{pct\_fw}       & share of XI minutes to forwards \\
        \texttt{depth\_sh}     & shot gap: starters vs substitutes \\
        \bottomrule
      \end{tabularx}
    \end{block}
    \vspace{0.3em}
      \scriptsize
      \setlength{\itemsep}{0pt}
      \setlength{\parskip}{0pt}
      \begin{itemize}
        \item Compute player rolling stats on a per-90 basis.
        \item Aggregate the starting XI by minutes played.
        \item Produce home, away, and net versions of each team stat.
      \end{itemize}
      \vspace{-0.2em}
      {\scriptsize
      \textbf{No data leakage:} only the five matches before the current game are used.\\[0.1em]
      \textbf{Form follows the player:} recent player form travels with transfers, so predictions rely on player form more than just current team affiliation.
      }

  \end{columns}
\end{frame}

\section{Feature Selection}
%% ═══════════════════════════════════════════════════════════

\begin{frame}{Feature Selection — Four-Method Protocol}
\presenterphoto{ioana_ml}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{1. Filter Method}
      \footnotesize\parbox[t][1.6cm][t]{\linewidth}{Simple idea: if I know this feature, do I get a better clue
      about the match result? Good for catching patterns that are not just straight lines.}
    \end{block}
    \vspace{0.15em}
    \begin{block}{2. Wrapper Method}
      \footnotesize\parbox[t][1.6cm][t]{\linewidth}{Start with many features, remove the weakest ones step by step,
      and keep the set that gives the best validation score.
      Best subset: \textbf{26 features}.}
    \end{block}

    \column{0.48\textwidth}
    \begin{block}{3. Embedded Method — Lasso}
      \footnotesize\parbox[t][1.6cm][t]{\linewidth}{A linear model with a penalty that pushes unhelpful features
      down to exactly zero. 17 features were removed; \textbf{0 player features} were.}
    \end{block}
    \vspace{0.15em}
    \begin{block}{4. Embedded Method — Decision Tree}
      \footnotesize\parbox[t][1.6cm][t]{\linewidth}{Let a forest of trees rank which features it uses most to make
      good splits. This gives a second opinion that broadly matches RFECV.}
    \end{block}
  \end{columns}
\end{frame}

%% ── Slide 3: Limits & Diagnostic ───────────────────────────────
\begin{frame}[t,shrink=8]{Linear Baseline — Limits \& Diagnostic}
\presenterphoto{ava_ml}
  \scriptsize \textbf{Why did adding features hurt?} Two compounding problems explain it.
  \vspace{0.3em}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \begin{block}{1. Multicollinearity}
      \scriptsize Team-level and player-level features measure overlapping things (a team's xG is essentially a function of its players' xG). Linear models cannot decide where to attribute the weight – the optimisation has multiple equivalent solutions, leading to \alert{unstable coefficients}.
    \end{block}
    \column{0.48\textwidth}
    \begin{block}{2. Curse of dimensionality}
      \scriptsize Only \alert{417 training matches} against up to \alert{76 features} – a samples-to-features ratio of $\sim$5.5. Rule of thumb: linear models need 10--20$\times$ more samples than features. Below that, they \textbf{overfit the noise} rather than learn the signal.
    \end{block}
  \end{columns}
  \vspace{0.3em}
  \begin{block}{This negative result is what motivated the tree models}
    \scriptsize Tree-based models handle correlated features natively – each split uses only one feature at a time, so they're \emph{locally low-dimensional}. \textbf{Tree models flip the trend} (RF: 1.26 $\to$ 1.06 $\to$ 1.05). Same data, same protocol, opposite outcome – which Samuel will detail next.
  \end{block}
  \vspace{0.2em}
  \scriptsize\textit{\textbf{Lesson:} model class matters more than feature count. The player-level signal exists – but linear models cannot surface it.}
\end{frame}

\begin{frame}[t,shrink=10]{Modeling — Hyperparameters \& Tuning}
\presenterphoto{sam_ml}
  \footnotesize
  Every model is tuned with \textbf{\texttt{RandomizedSearchCV}} (random configs drawn
  from each grid below), scored by \textbf{log-loss} (\texttt{neg\_log\_loss}), not
  accuracy. Cross-validation uses \textbf{\texttt{TimeSeriesSplit} (5 folds)} so the
  validation fold is always \alert{later in time} than the training fold --- data is never
  shuffled across the time boundary. Draw imbalance is handled with \textbf{balanced class
  weights} (RF) / \textbf{sample weights} (XGB); the best estimator is then
  \textbf{isotonic-calibrated} (\texttt{CalibratedClassifierCV}, temporal inner split)
  before the 50/50 RF\,+\,XGB ensemble.

  \vspace{0.4em}

\section{Key Findings}
%% ═══════════════════════════════════════════════════════════

\begin{frame}{Key Findings}
\presenterphoto{sam_ml}

\vspace{-0.25em}

{\scriptsize
\setlength{\parskip}{0pt}
\setlength{\itemsep}{0.25em}
\renewcommand{\baselinestretch}{0.92}\selectfont

\begin{enumerate}
  \item Player-level rolling statistics add real predictive signal beyond team form and bookmaker odds.

  \item Feature selection backs that up: \texttt{net\_pl\_starter\_mins} and
  \texttt{net\_pl\_pct\_fw} rank highly across all four methods, and RFECV keeps
  10 player features in its final subset.

  \item Tree models give the clearest proof of the player signal: as we move from
  team-only features to the full player regime, every tree model improves, while
  logistic regression gets worse under the extra correlation and dimensionality.

  \item The best tree result is the RF + XGBoost ensemble on the full player regime
  at \textbf{1.04} log-loss and 50.6\% accuracy, so ensembling still extracts a small final gain.

  \item KNN is a useful surprise: with $k{=}50$ it ties the stronger models
  (\textbf{1.01} in regime B, 1.02 in regime C), suggesting that averaging similar
  past match states works well when the sample is modest and noisy.

  \item The full pipeline is fully reproducible with only public data and closes about
  one-third of the gap from uniform guessing to Understat's 0.91 benchmark.
\end{enumerate}
}

\end{frame}
